 13 /**
 14  * @misc
 15  * @class <i>concrete</i> Facade implementation used to
 16  * facilitate the startup process of the MV&C.  This is
 17  * the 'hub' that accesses and communicates with the Proxies,
 18  * Mediators and Commands that do the work in your application.
 19  * <p>
 20  * Excerpts from "Implementation Idioms and Best Practices" by Cliff Hall:
 21  * "...By composition then, the Façade implements and exposes the
 22  * features of the Model, View and Controller; aggregating their
 23  * functionality and shielding the developer from direct interaction
 24  * with the Core actors of the framework..."</p>
 25  * <p> "By convention, it is named ‘ApplicationFacade’, but you
 26  * may call it whatever you like".</p>
 27  * <p> "Once the application’s View
 28  * hierarchy has been built, the PureMVC apparatus is
 29  * started and the Model and View regions are prepared for use."
 30  * For more information on creating the  concrete Facade, see
 31  * page 11 in "Implementation Idioms and Best Practices" by Cliff Hall
 32  * </p>
 33  * @extends Facade
 34  * @author Justin Wilaby
 35  */
 36 var ApplicationFacade = function(){
 37 
 38     /**
 39      * Required by MooTools for inheritance.
 40      * @type Class
 41      */
 42     this.Extends = new Class(new Facade());
 43 
 44     /**
 45      * The <code>Model</code> <code>View</code> and
 46      * <code>Controller</code> are initialized in a deliberate
 47      * order to ensure internal dependencies are satisfied before
 48      * operations are performed.<p>
 49      * <code>initializeController()</code> should be overridden
 50      * for the specific purpose of registering your commands. Any attempt to
 51      * register <code>Mediator</code>s here will result in an error.
 52      * being thrown because the View has not yet been initialized.</p>
 53      * <p>calling <code>this.parent()</code> is also required.
 54      */
 55     this.initializeController = function()
 56     {
 57 	// Always call this.parent()
 58 	this.parent();
 59 	this.registerCommand(ApplicationFacade.STARTUP, StartupCommand);
 60 	this.registerCommand(ApplicationFacade.RETRIEVE_CONFIG_OPTION, RetrieveConfigOptionCommand);
 61     }
 62 
 63     /**
 64      * Method used to start the system.  
 65      * 
 66      * @param {Shell} viewComponent the Shell instance used as the vew.
 67      */
 68     this.startup = function(viewComponent /* Shell */)
 69     {
 70 	this.sendNotification(ApplicationFacade.STARTUP, viewComponent);
 71     }
 72 }

337 /**
338  * @misc
339  * @class An ultra-light framework class used as a wrapper
340  * and psudo-extenstion of the MooTools Native Element
341  * implementation. Its primary goal is to encuorage the addition 
342  * of  elements to the DOM in a top-down fashion.
343  *
344  * @param {mixed} element String or Element to become the subject
345  * of ths instance.<p>If a string is passed, an element ID is assumed first.
346  * if the element ID is not found in the DOM, a tag name is assumed and
347  * MooTools will attempt to create that Element.</p><p>If an Element is
348  * passed in, it is left as-is.</p>
349  *
350  * @param {Object} properties The Object containing the default properties
351  * to be set on the Element.
352  */
353 var UIComponent = function(element /* mixed String or Element */, properties /* Object */){
354 
355     /**
356      * <code>true</code> if the component has
357      * been initialized <code>false</code> otherwise
358      * @type Boolean
359      */
360     this.initialized = false;
361 
362     /**
363      * The Native <code>Element</code> used as
364      * the subject of this UIComponent instance.
365      * @type Element
366      */
367     this.element = null;
368     
369     /**
370      * @ignore
371      */
372     this.initialize = function(element /* mixed String or Element */, properties /* Object */)
373     {
374 	this.element = $(element);
375 	if (!this.element)
376 	    this.element = new Element(element, properties);
377 	else
378 	    this.element.setProperties(properties);
379 	// Copy methods of the Element object to
380 	// 'this' and bind the functions to the element itself.
381 	// This creates a transparent wrapper in the UIComponent for
382 	// each method of the Element.
383 	var e = this.element;
384 	for (var key in e)
385 	{
386 	    var type = null;
387 	    try
388 	    {
389 		// IE 7+ has a problem with this sometimes.
390 		type = typeof e[key];
391 	    }
392 	    catch(e){}
393 	    if (type == "function" && !this[key])
394 	    {
395 		try
396 		{
397 		    // Safari has trouble here with some function binding
398 		    this[key] = e[key].bind(e);
399 		}
400 		catch(e){}
401 	    }
402 	}
403     }
404 
405     /**
406      * Used in the creation of children used in this component.
407      * Override this method to create new UIComponents or 'grab'
408      * existing DOM nodes.
409      */
410     this.initializeChildren = function(){}
411     /**
412      * Called after <code>initializeChildren()</code>.
413      * Override this method for additional processing
414      * or adding event listeners to children
415      */
416     this.childrenInitialized = function(){}
417     /**
418      * Called after <code>childrenInitialized()</code>
419      * Override this method to perform any final processing
420      * before the child is considered initialized.
421      */
422     this.initializationComplete = function()
423     {
424 	this.initialized = true;
425     }
426 
427     /**
428      * Wrapper for the MooTools <code>Element.grab()</code>
429      * that serves 2 primary purposes.
430      * <ol>
431      * <li>Appends the child UIComponent's DOMElement
432      * to this one.</li>
433      * <li>calls the child's initialization routines to gaurantee all
434      * elements are added to the DOM in a 'top down' fashion. (else IE will leak like a siv)</li>
435      * </ol>
436      * @param {UIComponent} child The child to append.
437      * @return {UIComponent} this object - can be used for chaining
438      */
439     this.addChild = function(child /* UIComponent */ )
440     {
441 	this.grab(child.element);
442 	// Initialize child
443 	child.initializeChildren();
444 	child.childrenInitialized();
445 	child.initializationComplete();
446 	// Fire an added event
447 	child.fireEvent('added');
448 	
449 	return this;
450     }
451 }

667     /**
668      * Creates buttons based on the array of <code>BoxConfigVO</code>
669      * objects.  In this case, this function is called from the <code>ControlPanelMediator</code>'s
670      * <code>onRegister()</code> method and is based on the data contained
671      * within the <code>ConfigProxy</code>'s <code>data</code> property.
672      *
673      * @param {BoxConfigVO[]} value An array of BoxConfigVO objects to base the config buttons on
674      * @see BoxConfigVO
675      * @see ControlPanelMediator
676      * @see ConfigProxy
677      */
678     this.setConfigurationButtons = function(value /* Array */)
679     {
680 	// Remove the old buttons if we have them
681 	var i = 0;
682 	var button = null;
683 	if (this.configButtons.length)
684 	{
685 	    i = this.configButtons.length;
686 	    while(i--)
687 	    {
688 		button = this.configButtons[i];
689 		button.removeEvent('click', this.button_clickHandler);
690 		button.dispose();
691 	    }
692 	    this.configButtons = [];
693 	}
694 	// Add the new ones based on the passed array
695 	var posY = 50;
696 	var len = value.length;
697 	for (i = 0; i < len; i++)
698 	{
699 	    var boxConfigVO = value[i];
700 	    button = new UIComponent('button', {html:boxConfigVO.id, styles:{position:"absolute", top:posY, left:5, width:150}});
701 	    // Store the index location of the button so
702 	    // we can easily find which BoxConfig to load.
703 	    button.store("index", i);
704 	    // Add the click handler.
705 	    button.addEvent('click', this.button_clickHandler);
706 	    // Add it to the DOM.
707 	    this.addChild(button);
708 	    // Retain it in our array for later reference.
709 	    this.configButtons.push(button);
710 	    // Position the button.
711 	    posY += button.getSize().y;
712 	}
713 	// Reposition the current config label so
714 	// that its below the buttons.
715 	this.currentConfigLabel.setStyle("top", posY.toString() + "px");
716 	posY += this.currentConfigLabel.getSize().y + 10; // Padding
717 	// Set the position of the start/stop button
718 	this.stopStartButton.setStyle("top", (posY + 10).toString() + "px");
719     }

994 	this.vp.x = this.getSize().x / 2;

1002     /**

1007      * basis for the <code>WorldSpace</code> state.

1012 	if (this.currentConfig == value)
1013 	    return;
1014 	this.currentConfig = value;
1015 	// Stop the timer if we have one

1017 	this.removeAllBoxes();
1018 	this.createWorldFromConfig();
1019 	this.startAnimation();
1020     }
1021 
1022     /**
1023      * Toggles the animation between start and stop
1024      * and returns true if the animation was started or
1025      * false if it was stopped.  This method is usually called by the
1026      * <code>WorldSpaceMediator</code> in response to the

1069     /**
1070      * @ignore
1071      * Creates instances of the <code>Box</code> based on
1072      * the current <code>BoxConfigVO</code>
1073      */
1074     this.createWorldFromConfig = function()
1075     {
1076 	if (!this.currentConfig)
1077 	    return;
1078 
1079 	this.focalLength =  this.currentConfig.focalLength;
1080 	var numBoxes = this.currentConfig.numBoxes;
1081 	var color = this.currentConfig.color;
1082 	var boxSize = this.currentConfig.boxSize;
1083 	// Randomize the placement of the boxes
1084 	// in the world space.
1085 	while(numBoxes--)
1086 	{
1087 	    var box = new Box(boxSize, boxSize);
1088 	    box.posX = Math.random() * 250 - 125;
1089 	    box.posY = Math.random() * 250 - 125;
1090 	    box.posZ = Math.random() * this.focalLength - (this.focalLength / 2);
1091 	    box.setStyles({"background":color, border:"1px solid #FFF"});
1092 	    this.boxes.push(box);
1093 	    this.addChild(box);
1094 	}
1095     }

1097     /**
1098      *	@ignore
1099      * Starts the Animation.
1100      */
1101     this.startAnimation = function()
1102     {
1103 	if (!this.boxes.length)
1104 	    return;
1105 	this.addEvent("mousemove", this.mouseMoveHandler);
1106 	this.timerId = this.tick.periodical(10, this);
1107 	this.inTween = true;
1108 	// Notify interested entities that the state has changed
1109 	this.fireEvent("animationStateChanged", this.inTween);
1110     }
1111 
1112     /**
1113      * @ignore
1114      * Updates the position, size, z-index and color
1115      * of each box.
1116      */
1117     this.tick = function()
1118     {
1119 	var i = this.boxes.length;
1120 	var rads = (this.angle * Math.PI) / 180;
1121 	var cos = Math.cos(rads);
1122 	var sin = Math.sin(rads);
1123 	while(i--)
1124 	{
1125 	    var box = this.boxes[i];
1126 	    var x1 = (box.posX * cos) - (box.posZ * sin);
1127 	    var z1 = (box.posZ * cos) + (box.posX * sin);
1128 	    var scale = this.focalLength / (this.focalLength + box.posZ);
1129 
1130 	    var progress = z1 > 0 ? z1 /  (this.focalLength / 2) : 0;
1131 	    var bgColor = this.interpolateColor(this.currentConfig.color, "#222222", progress);
1132 	    var borderColor = this.interpolateColor("#FFFFFF", "#010101", progress);
1133 	    box.posX = x1;
1134 	    box.posZ = z1;
1135 	    
1136 	    box.setStyles({
1137 		left:this.vp.x + (box.posX * scale),
1138 		top:this.vp.y + (box.posY * scale),
1139 		width:box.unscaledWidth * scale,
1140 		height:box.unscaledHeight * scale,
1141 		"background-color":bgColor,
1142 		"border-color":borderColor
1143 	    });
1144 	}
1145 	this.sortZ();
1146     }
1147 
1148     /**
1149      * @ignore
1150      * updates the z-order
1151      */
1152     this.sortZ = function()
1153     {
1154 	var copy = this.boxes.concat();
1155 	copy.sort(function(a, b)
1156 	{
1157 	    if(a.posZ < b.posZ)
1158 		return 1;
1159 	    if (a.posZ > b.posZ)
1160 		return -1;
1161 	    return 0;
1162 	});
1163 	var i = copy.length;
1164 	while(i--)
1165 	{
1166 	    var box = copy[i];
1167 	    box.setStyle("z-index", i+1);
1168 	}
1169     }
1170 
1171     /**
1172      *@ignore
1173      * color interpoation method
1174      */
1175     this.interpolateColor = function(originalColorHex /* String */, newColorHex /* String */, progress /* Number */)
1176     {
1177 	var originalColor = originalColorHex.hexToRgb(true); // Array of Number
1178 	var newColor = newColorHex.hexToRgb(true); // Array of Number
1179 
1180 	var diffRed = newColor[0] - originalColor[0];
1181 	var diffGreen = newColor[1] - originalColor[1];
1182 	var diffBlue = newColor[2] - originalColor[2];
1183 
1184 	var interpolatedRed = Math.round(originalColor[0]+(diffRed*progress));
1185 	var interpolatedGreen = Math.round(originalColor[1]+(diffGreen*progress));
1186 	var interpolatedBlue = Math.round(originalColor[2]+(diffBlue*progress));
1187 
1188 	return [interpolatedRed,interpolatedGreen,interpolatedBlue].rgbToHex();
1189     }

1201     }
