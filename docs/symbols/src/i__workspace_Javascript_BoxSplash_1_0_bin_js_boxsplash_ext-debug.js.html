  7   /**
  8    * @class <p>A <i>concrete</i> <code>Facade</code> implementation used to facilitate the startup
  9    *        process of the MVC. This is the 'hub' that accesses and communicates
 10    *        with the <code>Proxy</code>s, <code>Mediator</code>s and <code>Command</code>s that
 11    *        do the work in your application.</p>
 12    *        <p>
 13    *        Excerpts from "Implementation Idioms and Best Practices" by Cliff
 14    *        Hall: "...By composition then, the Facade implements and exposes the
 15    *        features of the Model, View and Controller; aggregating their
 16    *        functionality and shielding the developer from direct interaction with
 17    *        the Core actors of the framework..."
 18    *        </p>
 19    *        <p>
 20    *        "By convention, it is named <i>ApplicationFacade</i>, but you may call
 21    *        it whatever you like".
 22    *        </p>
 23    *        <p>
 24    *        "Once the application's View hierarchy has been built, the PureMVC
 25    *        apparatus is started and the Model and View regions are prepared for
 26    *        use." For more information on creating the concrete Facade, see page
 27    *        11 in "Implementation Idioms and Best Practices" by Cliff Hall.
 28    *        </p>
 29    * @extends Puremvc.patterns.Facade
 30    * 
 31    * @author Justin Wilaby
 32    * @author Tony DeFusco
 33    *
 34    * @constructs
 35    */
 36   constructor: function() {
 37     Boxsplash.ApplicationFacade.superclass.constructor.call(this);
 38   },
 39 
 40   /**
 41    * The <code>Model</code>, <code>View</code> and <code>Controller</code>
 42    * are initialized in a deliberate order to ensure internal dependencies are
 43    * satisfied before operations are performed.
 44    * <p>
 45    * <code>initializeController()</code> should be overridden for the specific
 46    * purpose of registering your commands. Any attempt to register
 47    * <code>Mediator</code>s here will result in an error being thrown
 48    * because the <code>View</code> has not yet been initialized.
 49    * </p>
 50    * <p>Calling <code>this.parent()</code> is also required.
 51    */
 52   initializeController: function() {
 53     // Always call parent
 54     Boxsplash.ApplicationFacade.superclass.initializeController.call(this);
 55     this.registerCommand(Boxsplash.ApplicationFacade.STARTUP, Boxsplash.controller.StartupCommand);
 56     this.registerCommand(Boxsplash.ApplicationFacade.RETRIEVE_CONFIG_OPTION, Boxsplash.controller.RetrieveConfigOptionCommand);
 57   },
 58 
 59   /**
 60    * Method used to start the system.
 61    *
 62    * @param {Boxsplash.view.components.Shell} viewComponent the <code>Shell</code> instance used as the <code>View</code>.
 63    */
 64   startup: function(viewComponent /* Shell */) {
 65     this.sendNotification(Boxsplash.ApplicationFacade.STARTUP, viewComponent);
 66   }
 67 });

367 Boxsplash.view.ControlPanelMediator = Ext.extend(Puremvc.patterns.Mediator, {
368   /**
369    * @class The <code>Mediator</code> subclass attached to the
370    * <code>ControlPanel</code> <code>View</code>. Its primary responsibilities here
371    * are to listen for and notify the system of user interactions that change
372    * the state of the application including that of other <code>View<code>s; specifically
373    * start/stop and config button clicks.  This <code>Mediator</code> is
374    * registered by the <code>ShellMediator</code>'s constructor.
375    *
376    * @extends Puremvc.patterns.Mediator
377    *
378    * @param {Boxsplash.view.components.ControlPanel} viewComponent the <code>ControlPanel</code> instance
379    * assigned to this mediator.
380    *
381    * @see Boxsplash.view.components.ControlPanel
382    * @see Boxsplash.view.ShellMediator
383    *
384    * @author Justin Wilaby
385    * @author Tony DeFusco
386    *
387    * @constructs
388    */
389   constructor: function(viewComponent /* ControlPanel */) {
390     Boxsplash.view.ControlPanelMediator.superclass.constructor.call(this, Boxsplash.view.ControlPanelMediator.NAME, viewComponent);
391     this.controlPanel = this.getViewComponent();
392     this.configProxy = this.facade.retrieveProxy(Boxsplash.model.ConfigProxy.NAME);
393 
394     // Replace listener handlers with methods bound to 'this'
395     this.loadConfigHandler = this.loadConfigHandler.createDelegate(this);
396     this.toggleStartStopHandler = this.toggleStartStopHandler.createDelegate(this);
397   },
398 
399   /**
400    * A named shortcut to the <code>ControlPanel</code> instance.  This
401    * prevents us from having to reference the more
402    * ambiguous <code>viewComponent</code> property.
403    * @type Boxsplash.view.components.ControlPanel
404    */
405   controlPanel: null,
406 
407   /**
408    * The <code>ConfigProxy</code> instance registered to the
409    * <code>ApplicationFacade</code>. It is retrieved here to act as
410    * a descriptor for the initial state of the <code>View</code> based on
411    * the <code>data</code> property contents.
412    *
413    * @type Boxsplash.model.ConfigProxy
414    * @see Boxsplash.ApplicationFacade
415    */
416   configProxy: null,
417 
418   /**
419    * Provides a list of notification interests to the <code>View</code>.
420    * Without an accurate list, the <code>handleNotification()</code> method
421    * may not be invoked.  A common mistake made by developers is to provide handling
422    * routines in the <code>handleNotification()</code> method but forget to
423    * add the notification name in the <code>listNotificationInterests()</code> array.
424    * <p>Note that changing this array at runtime will not have any effect on
425    * notification interests since this method is called by the <code>View</code>
426    * a single time when the <code>Mediator</code> is first registered.
427    *
428    * @return {String[]} the array of notification names to act upon.
429    */
430   listNotificationInterests: function() {
431     return [
432             Boxsplash.model.ConfigProxy.CONFIG_OPTION_RETRIEVED,
433             Boxsplash.ApplicationFacade.ANIMATION_STATE_CHANGED
434             ];
435   },
436 
437   /**
438    * Handles notifications broadcasted by the system provided that
439    * the <code>Notification</code> is listed in the <code>listNotificationInterests()</code>
440    * return value.
441    *
442    * @param {Puremvc.patterns.Notification} notification the notification to act upon.
443    */
444   handleNotification: function(notification /* Notification */) {
445     switch (notification.getName()) {
446       case Boxsplash.model.ConfigProxy.CONFIG_OPTION_RETRIEVED:
447         var boxConfigVO = notification.getBody();
448         this.controlPanel.setConfigurationLabel(boxConfigVO.id);
449         break;
450 
451       case Boxsplash.ApplicationFacade.ANIMATION_STATE_CHANGED:
452         this.controlPanel.animationStateChanged(notification.getBody());
453         break;
454     }
455   },
456 
457   /**
458    * Performs actions when the <code>Mediator</code> is registered by
459    * the <code>View</code>.  Here we listen for the "loadConfig" and "toggleStartStop"
460    * events from the <code>ControlPanel</code> and set the configuration buttons
461    * based on the contents of the <code>ConfigProxy</code>'s <code>data</code>
462    * property.
463    *
464    * @see Boxsplash.view.components.ControlPanel
465    * @see Boxsplash.model.ConfigProxy
466    */
467   onRegister: function() {
468     Boxsplash.view.ControlPanelMediator.superclass.onRegister.call(this);
469     this.controlPanel.addListener("loadConfig", this.loadConfigHandler);
470     this.controlPanel.addListener("toggleStartStop", this.toggleStartStopHandler);
471     this.controlPanel.setConfigurationButtons(this.configProxy.data);
472   },
473 
474   /**
475    * Removes listeners so that the <code>View</code> can be properly disposed of
476    * and garbage collected.
477    */
478   onRemove: function() {
479     Boxsplash.view.ControlPanelMediator.superclass.onRemove.call(this);
480     this.controlPanel.removeListener("loadConfig", this.loadConfigHandler);
481     this.controlPanel.removeListener("toggleStartStop", this.toggleStartStopHandler);
482   },
483 
484   /**
485    * Event handler for the "loadConfig" event dispatched by the
486    * <code>ControlPanel</code> in response to user clicks on any
487    * of the configuration buttons.
488    *
489    * @param {int} index the index of the desired configuration option settings to load.
490    */
491   loadConfigHandler: function(index) {
492     this.sendNotification(Boxsplash.ApplicationFacade.RETRIEVE_CONFIG_OPTION, index);
493   },
494 
495   /**
496    * Event handler for the "toggleStartStop" event dispatched
497    * by the <code>ControlPanel</code> in response to user clicks
498    * on the <i>startStop</i> button.
499    */
500   toggleStartStopHandler: function() {
501     this.sendNotification(Boxsplash.ApplicationFacade.TOGGLE_START_STOP);
502   }
503 });

1351   /**

1353    * of each box for each "tick" of the timer.

1356     var i = this.boxes.length;
1357     var rads = (this.angle * Math.PI) / 180;
1358     var cos = Math.cos(rads);
1359     var sin = Math.sin(rads);
1360     while(i--) {
1361       var box = this.boxes[i];
1362       var x1 = (box.posX * cos) - (box.posZ * sin);
1363       var z1 = (box.posZ * cos) + (box.posX * sin);
1364       var scale = this.focalLength / (this.focalLength + box.posZ);
1365 
1366       var progress = (z1 > 0) ? z1 /  (this.focalLength / 2) : 0;
1367       var bgColor = this.interpolateColor(this.currentConfig.color, "#222222", progress);
1368       var borderColor = this.interpolateColor("#FFFFFF", "#010101", progress);
1369       box.posX = x1;
1370       box.posZ = z1;
1371 
1372       box.element.setStyle({
1373         left: this.vp.x  + (box.posX * scale),
1374         top: this.vp.y + (box.posY * scale),
1375         width: box.unscaledWidth * scale,
1376         height: box.unscaledHeight * scale,
1377         "background-color": bgColor,
1378         "border-color": borderColor
1379       });
1380 
1381     }

1403   /**
1404    * A routine for interpolating the appropriate color a given distance between two given colors.
1405    *
1406    * @param originalColorHex {String} the RGB color value of the start color.
1407    * @param newColorHex {String} the RGB color value for the end color.
1408    * @param progress {Number} the ratio between the start and end colors.
1409    *
1410    * @return the RGB color value for the color that occurs between the given start and end colors.
1411    * @type String
1412    */
1413   interpolateColor: function(originalColorHex /* String */, newColorHex /* String */, progress /* Number */) {
1414     var originalColor = parseInt(originalColorHex.replace('#', '0x'));
1415     var newColor = parseInt(newColorHex.replace('#', '0x'));
1416 
1417     // Extracts the red channel.
1418     var origRed = (originalColor & 0xFF0000) >> 16;
1419 
1420     // Extracts the green channel.
1421     var origGreen = (originalColor & 0xFF00) >> 8;
1422 
1423     // Extracts the blue channel.
1424     var origBlue = originalColor & 0xFF;
1425     // --------------------------------
1426 
1427     // Extracts the red channel.
1428     var newRed = (newColor & 0xFF0000) >> 16;
1429 
1430     // Extracts the green channel.
1431     var newGreen = (newColor & 0xFF00) >> 8;
1432 
1433     // Extracts the blue channel.
1434     var newBlue = newColor & 0xFF;
1435     //---------------------------------
1436 
1437     var diffRed = newRed - origRed;
1438     var diffGreen = newGreen - origGreen;
1439     var diffBlue = newBlue - origBlue;
1440 
1441     var interpolatedRed = origRed + (diffRed * progress);
1442     var interpolatedGreen = origGreen + (diffGreen * progress);
1443     var interpolatedBlue = origBlue + (diffBlue * progress);
1444 
1445     var payload = Math.abs((interpolatedRed << 16) | (interpolatedGreen << 8) | interpolatedBlue);
1446 
1447     return "#" + this.rgbToHex([((payload & 0xFF0000) >> 16), (payload & 0xFF00) >> 8,  payload & 0xFF]);
1448   },

1469    * @param value {Number[]} an array of RGB color components to convert.

1475     /**
1476      * Converts a decimal value to its corresponding hex value as a string.
1477      * @inner
1478      * @param N the decimal value to convert to a hex string.
1479      */
1480     function _toHex(N) {
1481       if (N == null) {
1482         return "00";
1483       }
1484       N = parseInt(N);
1485       if (N == 0 || isNaN(N)) {
1486         return "00";
1487       }
1488       N = Math.max(0, N);
1489       N = Math.min(N, 255);
1490       N = Math.round(N);
1491       return "0123456789ABCDEF".charAt((N-N % 16) / 16)
1492           + "0123456789ABCDEF".charAt(N % 16);
1493     };
1494     var R = value[0], G = value[1], B = value[2];
1495     return _toHex(R) + _toHex(G) + _toHex(B);
